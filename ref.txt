                    }

                    contactForm.querySelectorAll('.form-input').forEach(el => {
                        if (!el.classList.contains('border-red-coral')) {
                            el.classList.add('border-emerald-green');
                        }
                    });

                    if(submitButton) submitButton.disabled = true;
                    if(buttonText) buttonText.textContent = 'Sending...';
                    if(buttonIcon) buttonIcon.classList.add('hidden');
                    if(formStatusEl) formStatusEl.textContent = '';

                    setTimeout(() => {
                        if(submitButton) {
                            submitButton.classList.remove('bg-accent', 'dark:bg-dark-accent');
                            submitButton.classList.add('!bg-emerald-green', 'text-white');
                        }
                        if(buttonText) buttonText.textContent = 'Message Sent!';
                        if(formStatusEl) {
                            formStatusEl.textContent = 'Thank you for your message!';
                            formStatusEl.classList.add('text-emerald-green');
                        }
                        contactForm.reset();
                        contactForm.querySelectorAll('.form-input').forEach(el => el.classList.remove('border-emerald-green', 'border-red-coral', 'ring-red-coral'));

                        setTimeout(() => {
                            if(submitButton) {
                                submitButton.disabled = false;
                                submitButton.classList.remove('!bg-emerald-green', 'text-white');
                                submitButton.classList.add('bg-accent', 'dark:bg-dark-accent');
                            }
                            if(buttonText) buttonText.textContent = 'Send Message';
                            if(buttonIcon) buttonIcon.classList.remove('hidden');
                            if(formStatusEl) {
                                formStatusEl.textContent = '';
                                formStatusEl.className = 'mt-2 text-center text-sm font-body';
                            }
                        }, 3000);

                    }, 1500);
                });
             }

            // --- Footer Current Year ---
            if (currentYearEl) currentYearEl.textContent = new Date().getFullYear();

            // --- Scroll Progress Bar ---
            window.addEventListener('scroll', () => {
                const scrollableHeight = document.documentElement.scrollHeight - window.innerHeight;
                const scrolled = window.scrollY;
                const progress = scrollableHeight > 0 ? (scrolled / scrollableHeight) : 0;
                if(scrollProgressBar) scrollProgressBar.style.transform = `scaleX(${progress})`;
            });

            // --- Scroll Reveal Animations (with Staggering) ---
            const revealObserverOptions = { threshold: 0.1, rootMargin: "0px 0px -50px 0px" };
            const revealCallback = (entries, observer) => {
                entries.forEach(entry => {
                    if (entry.isIntersecting) {
                        const delay = entry.target.dataset.revealDelay || 0; // Get delay from data attribute
                        entry.target.style.transitionDelay = `${delay}ms`; // Apply delay

                        entry.target.classList.add('reveal-visible');
                        if (entry.target.classList.contains('reveal-slide-left') || entry.target.classList.contains('reveal-slide-right')) {
                            entry.target.classList.add('slide-in');
                        }
                        observer.unobserve(entry.target); // Unobserve after revealing
                    }
                });
            };
            const revealObserver = new IntersectionObserver(revealCallback, revealObserverOptions);
            // Observe all elements with reveal classes or data-reveal-delay
            document.querySelectorAll('.reveal-hidden, .reveal-slide-left, .reveal-slide-right, [data-reveal-delay]').forEach(el => {
                // Ensure base class is present if only delay is set
                if (!el.classList.contains('reveal-hidden') && !el.classList.contains('reveal-slide-left') && !el.classList.contains('reveal-slide-right')) {
                   el.classList.add('reveal-hidden');
                }
                 revealObserver.observe(el);
            });

            // --- Active Navigation Link Highlighting ---
            let currentActiveSectionId = null;
            const navObserverOptions = { root: null, rootMargin: "-40% 0px -60% 0px", threshold: 0 };
            const navObserverCallback = (entries) => {
                 let intersectingSectionId = currentActiveSectionId;

                 entries.forEach(entry => {
                     if (entry.isIntersecting && entry.intersectionRatio > 0) {
                         intersectingSectionId = entry.target.getAttribute('id');
                     }
                 });

                 if (intersectingSectionId !== currentActiveSectionId) {
                    currentActiveSectionId = intersectingSectionId;
                    navLinks.forEach(link => {
                        link.classList.toggle('nav-active', link.getAttribute('data-section') === currentActiveSectionId);
                    });
                    mobileNavLinks.forEach(link => {
                         link.classList.toggle('nav-active', link.getAttribute('data-section') === currentActiveSectionId);
                    });
                 }
            };
            const navObserver = new IntersectionObserver(navObserverCallback, navObserverOptions);
            sections.forEach(section => { if(section) navObserver.observe(section); });
            // --- Initial Nav Highlight on Load ---
            const initialHash = window.location.hash.substring(1);
            if (initialHash) {
                currentActiveSectionId = initialHash;
                 navLinks.forEach(link => {
                     link.classList.toggle('nav-active', link.getAttribute('data-section') === currentActiveSectionId);
                 });
                 mobileNavLinks.forEach(link => {
                      link.classList.toggle('nav-active', link.getAttribute('data-section') === currentActiveSectionId);
                 });
            }

            // --- Clipboard Copy Functionality ---
            let toastTimeout;
            copyButtons.forEach(button => {
                button.addEventListener('click', () => {
                    const textToCopy = button.dataset.copyText;
                    if (textToCopy && navigator.clipboard) {
                        navigator.clipboard.writeText(textToCopy).then(() => {
                            if (copyToast) {
                                copyToast.classList.add('show');
                                clearTimeout(toastTimeout);
                                toastTimeout = setTimeout(() => {
                                    copyToast.classList.remove('show');
                                }, 2000);
                            }
                        }).catch(err => {
                            console.error('Failed to copy text: ', err);
                        });
                    }
                });
            });

            // --- Animated Counters ---
            const animateCounter = (el) => {
                const target = parseInt(el.dataset.target, 10);
                if (isNaN(target)) return; // Skip if target is not a number
                let current = 0;
                const duration = 1500; // Animation duration in ms
                const stepTime = 20; // Time between steps in ms
                const totalSteps = duration / stepTime;
                const increment = target / totalSteps;

                const timer = setInterval(() => {
                    current += increment;
                    if (current >= target) {
                        el.textContent = target; // Ensure final value is exact
                        clearInterval(timer);
                    } else {
                        el.textContent = Math.ceil(current); // Display integer value
                    }
                }, stepTime);
            };

            const counterObserverOptions = { threshold: 0.5 }; // Trigger when 50% visible
            const counterObserverCallback = (entries, observer) => {
                entries.forEach(entry => {
                    if (entry.isIntersecting) {
                        animateCounter(entry.target);
                        observer.unobserve(entry.target); // Animate only once
                    }
                });
            };
            const counterObserver = new IntersectionObserver(counterObserverCallback, counterObserverOptions);
            counterElements.forEach(el => counterObserver.observe(el));

        }); // End DOMContentLoaded
    </script>

</body>
